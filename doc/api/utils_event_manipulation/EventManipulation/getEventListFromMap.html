static List<Event> getEventListFromMap(HashMap map, String uc,
      [String turno = "N/A"]) {
    if (turno != "N/A") return map[uc][turno];
    List<Event> events = [];
    map[uc].forEach((_, v) {
      events.addAll(v);
    });
    return events;
  }
