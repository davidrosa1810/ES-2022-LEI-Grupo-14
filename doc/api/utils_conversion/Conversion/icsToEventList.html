static EventsFile icsToEventList(String ics) {
  int missingData = 0;
  List<Event> lista = [];
  ICalendar icd = ICalendar.fromString(ics);
  for (var ele in icd.data) {
    var obj = icdToEvent(ele);
    if (obj == Null) {
      missingData++;
    } else {
      lista.add(obj as Event);
    }
  }
  return EventsFile("", lista, missingData);
}
