factory Event.fromCSV(String csv) {
  var event = csv.split(RegExp(',(?=([^"]*"[^"]*")*[^"]*\$)'));

  return Event(
      event[0].replaceAll('"', ""),
      event[1].replaceAll('"', ""),
      event[2].replaceAll('"', ""),
      event[3].replaceAll('"', ""),
      event[4].replaceAll('"', ""),
      event[5].replaceAll('"', ""),
      event[6].replaceAll('"', ""),
      event[7].replaceAll('"', ""),
      event[8].replaceAll('"', ""),
      event[9].replaceAll('"', ""),
      event[10].replaceAll('"', "").trim());
}
